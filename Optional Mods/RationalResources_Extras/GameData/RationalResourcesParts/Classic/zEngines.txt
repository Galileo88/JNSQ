// Convert RR nuclear turbojet and rocket engines
@PART[RREngineNTJ1]:AFTER[RationalResources]
{
	@MODULE[ModuleEnginesFX]
	{
		@PROPELLANT[IntakeAtm]
		{
			@name = Atmosphere
		}
		@PROPELLANT[ThermalPower]
		{
			@name = CoreHeat
		}
	}
	@RESOURCE[EnrichedUranium]
	{
		@name = NuclearFuel
	}
	RESOURCE[DepletedFuel]
	{
		@name = NuclearWaste
	}
	RESOURCE[ThermalPower]
	{
		@name = CoreHeat
	}
	@MODULE[ModuleResourceConverter]
	{
		@INPUT_RESOURCE[EnrichedUranium]
		{
			@ResourceName = NuclearFuel
		}
		@OUTPUT_RESOURCE[DepletedFuel]
		{
			@ResourceName = NuclearWaste
		}
		@OUTPUT_RESOURCE[ThermalPower]
		{
			@ResourceName = CoreHeat
		}
	}
}
